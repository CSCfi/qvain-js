const object1 = {};

Object.defineProperty(object1, 'property1', {
  value: ['a'],
  writable: false,
  configurable: true,
  enumerable: true,
});

object1.property1.push('b');
object1.property1.splice(0, object1.property1.length)
// throws an error in strict mode

for (let prop in object1) {
  console.log(prop)
}

console.log(object1.property1);
// expected output: 42
